User Message Extraction

const userMessage = element.querySelector('[data-testid="user-message"]'); const content = userMessage?.innerText?.trim() || '';

Assistant Response with Reasoning (Claude 3.7+)

// New-style reasoning const thoughtHeader = element.querySelector('button.group\/status'); const statusTitle = thoughtHeader?.querySelector('.truncate')?.innerText?.trim(); // Extract reasoning steps const thoughtContent = element.querySelector('.flex-col.font-ui'); const stepContainers = thoughtContent?.querySelectorAll('.flex.flex-col.shrink-0'); const reasoningText = Array.from(stepContainers || []) .map(container => { const markdown = container.querySelector('.standard-markdown'); return markdown?.innerText?.trim() || ''; }) .filter(Boolean) .join('\n\n');

const sideBlocks = element.querySelectorAll('.border-border-300.rounded-lg'); sideBlocks.forEach(block => { const title = block.querySelector('button')?.innerText?.trim(); const content = block.querySelector('.standard-markdown')?.innerText?.trim(); // Process title and content for side-channel preservation });

Main Response Content

const claudeResponse = element.querySelector('.font-claude-response'); const mainMarkdown = Array.from(claudeResponse.querySelectorAll('.standard-markdown')) .filter(el => !el.closest('.border-border-300.rounded-lg')) .map(el => el.innerText?.trim()) .join('\n\n');

const artifactContainer = container.querySelector('[data-testid="artifact-container"], .artifact-block-cell'); const hasArtifact = !!artifactContainer; const artifactType = artifactContainer?.getAttribute('data-artifact-type'); const artifactLanguage = artifactContainer?.getAttribute('data-language');
